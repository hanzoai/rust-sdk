1//! Tool registry and execution for agents
2//!
3//! This module provides a unified tool registry that can integrate:
4//! - Built-in tools (file operations, search, etc.)
5//! - MCP server tools (via hanzo-mcp)
6//! - Custom tool callbacks
7
8use crate::traits::{AgentError, Result, ToolDefinition};
9use serde::{Deserialize, Serialize};
10use std::collections::HashMap;
11use std::sync::Arc;
12use tokio::sync::RwLock;
13
14/// Callback type for tool execution
15pub type ToolCallback = Arc<
16    dyn Fn(serde_json::Value) -> futures::future::BoxFuture<'static, Result<String>> + Send + Sync,
17>;
18
19/// A registered tool with its callback
20pub struct RegisteredTool {
21    pub definition: ToolDefinition,
22    pub callback: ToolCallback,
23    pub source: ToolSource,
24}
25
26/// Source of a tool (for debugging and management)
27#[derive(Debug, Clone, Serialize, Deserialize)]
28pub enum ToolSource {
29    /// Built-in tool
30    BuiltIn,
31    /// Tool from an MCP server
32    Mcp { server_name: String },
33    /// Custom tool registered at runtime
34    Custom { source: String },
35}
36
37/// Configuration for an MCP server
38#[derive(Debug, Clone, Serialize, Deserialize)]
39pub struct McpServerConfig {
40    /// Server name (unique identifier)
41    pub name: String,
42    /// Connection type
43    pub connection: McpConnection,
44    /// Environment variables for process-based servers
45    pub env: Option<HashMap<String, String>>,
46    /// Timeout in seconds
47    pub timeout_secs: Option<u64>,
48}
49
50/// MCP connection types
51#[derive(Debug, Clone, Serialize, Deserialize)]
52#[serde(tag = "type")]
53pub enum McpConnection {
54    /// HTTP-based connection
55    #[serde(rename = "http")]
56    Http { url: String },
57    /// WebSocket connection
58    #[serde(rename = "websocket")]
59    WebSocket { url: String },
60    /// Subprocess (stdio)
61    #[serde(rename = "process")]
62    Process {
63        command: String,
64        args: Option<Vec<String>>,
65    },
66}
67
68/// Central registry for all tools available to agents
69pub struct ToolRegistry {
70    tools: RwLock<HashMap<String, RegisteredTool>>,
71    mcp_servers: RwLock<HashMap<String, McpServerConfig>>,
72}
73
74impl ToolRegistry {
75    /// Create a new empty registry
76    pub fn new() -> Self {
77        Self {
78            tools: RwLock::new(HashMap::new()),
79            mcp_servers: RwLock::new(HashMap::new()),
80        }
81    }
82
83    /// Create a registry with default built-in tools
84    pub fn with_defaults() -> Self {
85        let registry = Self::new();
86        // Built-in tools would be registered here
87        // For now, keeping it empty as tools will come from MCP servers
88        registry
89    }
90
91    /// Register a custom tool
92    pub async fn register(
93        &self,
94        definition: ToolDefinition,
95        callback: ToolCallback,
96        source: ToolSource,
97    ) {
98        let name = definition.name.clone();
99        let tool = RegisteredTool {
100            definition,
101            callback,
102            source,
103        };
104        self.tools.write().await.insert(name, tool);
105    }
106
107    /// Register an MCP server and discover its tools
108    pub async fn register_mcp_server(&self, config: McpServerConfig) -> Result<Vec<String>> {
109        let server_name = config.name.clone();
110
111        // Store the config
112        self.mcp_servers
113            .write()
114            .await
115            .insert(server_name.clone(), config.clone());
116
117        // Discover and register tools from the server
118        let tool_names = self.discover_mcp_tools(&config).await?;
119
120        Ok(tool_names)
121    }
122
123    /// Discover tools from an MCP server
124    async fn discover_mcp_tools(&self, config: &McpServerConfig) -> Result<Vec<String>> {
125        use hanzo_mcp::mcp_methods;
126
127        let tools = match &config.connection {
128            McpConnection::Http { url } => mcp_methods::list_tools_via_http(url, None)
129                .await
130                .map_err(|e| AgentError::McpError(e.message))?,
131            McpConnection::Process { command, args } => {
132                let cmd_str = if let Some(args) = args {
133                    format!("{} {}", command, args.join(" "))
134                } else {
135                    command.clone()
136                };
137                mcp_methods::list_tools_via_command(&cmd_str, config.env.clone())
138                    .await
139                    .map_err(|e| AgentError::McpError(e.message))?
140            }
141            McpConnection::WebSocket { url: _ } => {
142                // WebSocket support would go here
143                return Err(AgentError::ConfigError(
144                    "WebSocket MCP not yet implemented".to_string(),
145                ));
146            }
147        };
148
149        let mut registered_names = Vec::new();
150        let server_name = config.name.clone();
151
152        for tool in tools {
153            // Prefix tool name with server name to avoid conflicts
154            let prefixed_name = format!("{}:{}", server_name, tool.name);
155
156            let definition = ToolDefinition {
157                name: prefixed_name.clone(),
158                description: tool.description.unwrap_or_default().to_string(),
159                parameters: serde_json::Value::Object((*tool.input_schema).clone()),
160                requires_confirmation: false,
161            };
162
163            // Create callback that routes to the MCP server
164            let config_clone = config.clone();
165            let tool_name = tool.name.clone();
166
167            let callback: ToolCallback = Arc::new(move |args: serde_json::Value| {
168                let config = config_clone.clone();
169                let name = tool_name.clone();
170                Box::pin(async move {
171                    let params = args.as_object().cloned().unwrap_or_default();
172                    execute_mcp_tool(&config, &name, params).await
173                })
174            });
175
176            self.register(
177                definition,
178                callback,
179                ToolSource::Mcp {
180                    server_name: server_name.clone(),
181                },
182            )
183            .await;
184
185            registered_names.push(prefixed_name);
186        }
187
188        Ok(registered_names)
189    }
190
191    /// Get a tool by name
192    pub async fn get(&self, name: &str) -> Option<ToolDefinition> {
193        self.tools
194            .read()
195            .await
196            .get(name)
197            .map(|t| t.definition.clone())
198    }
199
200    /// Execute a tool by name
201    pub async fn execute(&self, name: &str, args: serde_json::Value) -> Result<String> {
202        let tools = self.tools.read().await;
203        let tool = tools.get(name).ok_or_else(|| AgentError::ToolError {
204            tool_name: name.to_string(),
205            message: "Tool not found".to_string(),
206        })?;
207
208        (tool.callback)(args).await
209    }
210
211    /// List all available tools
212    pub async fn list(&self) -> Vec<ToolDefinition> {
213        self.tools
214            .read()
215            .await
216            .values()
217            .map(|t| t.definition.clone())
218            .collect()
219    }
220
221    /// List tools by source
222    pub async fn list_by_source(&self, source_filter: &str) -> Vec<ToolDefinition> {
223        self.tools
224            .read()
225            .await
226            .values()
227            .filter(|t| match &t.source {
228                ToolSource::BuiltIn => source_filter == "builtin",
229                ToolSource::Mcp { server_name } => server_name == source_filter,
230                ToolSource::Custom { source } => source == source_filter,
231            })
232            .map(|t| t.definition.clone())
233            .collect()
234    }
235}

243/// Execute a tool on an MCP server
244async fn execute_mcp_tool(
245    config: &McpServerConfig,
246    tool_name: &str,
247    params: serde_json::Map<String, serde_json::Value>,
248) -> Result<String> {
249    use hanzo_mcp::mcp_methods;
250
251    let result = match &config.connection {
252        McpConnection::Http { url } => {
253            mcp_methods::run_tool_via_http(url.clone(), tool_name.to_string(), params)
254                .await
255                .map_err(|e| AgentError::McpError(e.message))?
256        }
257        McpConnection::Process { command, args } => {
258            let cmd_str = if let Some(args) = args {
259                format!("{} {}", command, args.join(" "))
260            } else {
261                command.clone()
262            };
263            mcp_methods::run_tool_via_command(
264                cmd_str,
265                tool_name.to_string(),
266                config.env.clone().unwrap_or_default(),
267                params,
268            )
269            .await
270            .map_err(|e| AgentError::McpError(e.message))?
271        }
272        McpConnection::WebSocket { url: _ } => {
273            return Err(AgentError::ConfigError(
274                "WebSocket MCP not yet implemented".to_string(),
275            ));
276        }
277    };
278
279    // Extract text content from the result
280    let text = result
281        .content
282        .into_iter()
283        .filter_map(|c| c.as_text().map(|t| t.text.clone()))
284        .collect::<Vec<_>>()
285        .join("\n");
286
287    Ok(text)
288}

301    #[tokio::test]
302    async fn test_custom_tool_registration() {
303        let registry = ToolRegistry::new();
304
305        let definition = ToolDefinition::new("test_tool", "A test tool");
306        let callback: ToolCallback =
307            Arc::new(|_args| Box::pin(async { Ok("test result".to_string()) }));
308
309        registry
310            .register(
311                definition,
312                callback,
313                ToolSource::Custom {
314                    source: "test".to_string(),
315                },
316            )
317            .await;
318
319        let tools = registry.list().await;
320        assert_eq!(tools.len(), 1);
321        assert_eq!(tools[0].name, "test_tool");
322
323        let result = registry
324            .execute("test_tool", serde_json::json!({}))
325            .await
326            .unwrap();
327        assert_eq!(result, "test result");
328    }
