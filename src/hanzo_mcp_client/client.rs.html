1//! Unified MCP Client for multi-server orchestration
2//!
3//! This module provides a high-level client that manages connections to multiple
4//! MCP servers simultaneously, with support for:
5//! - Multiple transport protocols (HTTP, SSE, Process)
6//! - Automatic tool discovery and registration
7//! - Concurrency control via semaphore
8//! - Per-tool timeouts
9//! - Tool name prefixing to avoid conflicts
10

47/// Unified MCP client that manages connections to multiple MCP servers
48///
49/// # Features
50///
51/// - **Multi-server Management**: Connects to and manages multiple MCP servers simultaneously
52/// - **Automatic Tool Discovery**: Discovers available tools from connected servers
53/// - **Tool Name Prefixing**: Prefixes tool names to avoid conflicts between servers
54/// - **Concurrency Control**: Limits concurrent tool calls via semaphore
55/// - **Timeout Support**: Per-tool execution timeouts
56///
57/// # Example
58///
59/// ```rust,ignore
60/// use hanzo_mcp_client::McpClient;
61/// use hanzo_mcp_core::{McpClientConfig, McpServerConfig, McpServerSource};
62///
63/// #[tokio::main]
64/// async fn main() -> Result<(), Box<dyn std::error::Error>> {
65///     let config = McpClientConfig {
66///         servers: vec![
67///             McpServerConfig {
68///                 name: "local".to_string(),
69///                 source: McpServerSource::Process {
70///                     command: "mcp-server".to_string(),
71///                     args: vec![],
72///                     work_dir: None,
73///                     env: None,
74///                 },
75///                 ..Default::default()
76///             },
77///         ],
78///         max_concurrent_calls: Some(5),
79///         ..Default::default()
80///     };
81///
82///     let mut client = McpClient::new(config);
83///     client.initialize().await?;
84///
85///     let tools = client.list_tools().await;
86///     println!("Found {} tools", tools.len());
87///
88///     Ok(())
89/// }
90/// ```
91pub struct McpClient {
92    config: McpClientConfig,
93    servers: RwLock<HashMap<String, ServerConnection>>,
94    tools: RwLock<HashMap<String, McpToolInfo>>,
95    concurrency_semaphore: Arc<Semaphore>,
96}

98impl McpClient {
99    /// Create a new MCP client with the given configuration
100    pub fn new(config: McpClientConfig) -> Self {
101        let max_concurrent = config.max_concurrent_calls.unwrap_or(10);
102        Self {
103            config,
104            servers: RwLock::new(HashMap::new()),
105            tools: RwLock::new(HashMap::new()),
106            concurrency_semaphore: Arc::new(Semaphore::new(max_concurrent)),
107        }
108    }

110    /// Initialize connections to all configured servers
111    ///
112    /// This method connects to each enabled server and discovers available tools.
113    pub async fn initialize(&mut self) -> Result<()> {
114        for server_config in &self.config.servers {
115            if !server_config.enabled {
116                debug!("Skipping disabled server: {}", server_config.name);
117                continue;
118            }
119
120            match self.connect_server(server_config.clone()).await {
121                Ok(()) => {
122                    info!("Connected to MCP server: {}", server_config.name);
123                }
124                Err(e) => {
125                    error!(
126                        "Failed to connect to MCP server {}: {}",
127                        server_config.name, e
128                    );
129                    // Continue with other servers
130                }
131            }
132        }
133
134        if self.config.auto_register_tools {
135            self.discover_tools().await?;
136        }
137
138        Ok(())
139    }

141    /// Connect to a single server and discover its tools
142    async fn connect_server(&self, config: McpServerConfig) -> Result<()> {
143        let tools = self.list_server_tools(&config).await?;
144
145        let mut servers = self.servers.write().await;
146        servers.insert(config.id.clone(), ServerConnection { config, tools });
147
148        Ok(())
149    }

151    /// List tools from a specific server
152    async fn list_server_tools(&self, config: &McpServerConfig) -> Result<Vec<McpToolInfo>> {
153        let raw_tools = match &config.source {
154            McpServerSource::Http { url, .. } => {
155                mcp_methods::list_tools_via_http(url, None).await?
156            }
157            McpServerSource::Sse { url, .. } => mcp_methods::list_tools_via_sse(url, None).await?,
158            McpServerSource::Process {
159                command, args, env, ..
160            } => {
161                let cmd_str = if args.is_empty() {
162                    command.clone()
163                } else {
164                    format!("{} {}", command, args.join(" "))
165                };
166                mcp_methods::list_tools_via_command(&cmd_str, env.clone()).await?
167            }
168            McpServerSource::WebSocket { .. } => {
169                // WebSocket not yet implemented in rmcp-based client
170                warn!(
171                    "WebSocket transport not yet supported, skipping server: {}",
172                    config.name
173                );
174                return Ok(Vec::new());
175            }
176        };
177
178        // Convert to McpToolInfo with server context
179        let tools: Vec<McpToolInfo> = raw_tools
180            .into_iter()
181            .map(|t| McpToolInfo {
182                name: t.name.to_string(),
183                description: t.description.map(|d| d.to_string()),
184                input_schema: serde_json::Value::Object((*t.input_schema).clone()),
185                server_id: config.id.clone(),
186                server_name: config.name.clone(),
187            })
188            .collect();
189
190        Ok(tools)
191    }
192
193    /// Discover and register tools from all connected servers
194    async fn discover_tools(&self) -> Result<()> {
195        let servers = self.servers.read().await;
196        let mut all_tools = self.tools.write().await;
197
198        for (server_id, connection) in servers.iter() {
199            for tool in &connection.tools {
200                // Apply tool prefix if configured
201                let tool_name = if let Some(prefix) = &connection.config.tool_prefix {
202                    format!("{}_{}", prefix, tool.name)
203                } else {
204                    tool.name.clone()
205                };
206
207                let mut tool_info = tool.clone();
208                tool_info.name = tool_name.clone();
209                all_tools.insert(tool_name, tool_info);
210            }
211            debug!(
212                "Registered {} tools from server {}",
213                connection.tools.len(),
214                server_id
215            );
216        }
217
218        info!("Total tools registered: {}", all_tools.len());
219        Ok(())
220    }

242    /// Call a tool by its prefixed name
243    ///
244    /// This method handles routing to the correct server, concurrency control,
245    /// and timeout enforcement.
246    pub async fn call_tool(
247        &self,
248        tool_name: &str,
249        arguments: serde_json::Value,
250    ) -> Result<ToolResult> {
251        // Find the tool and its server
252        let (original_name, server_config) = {
253            let tools = self.tools.read().await;
254            let tool_info = tools
255                .get(tool_name)
256                .ok_or_else(|| McpError::new(format!("Tool not found: {}", tool_name)))?;
257
258            let servers = self.servers.read().await;
259            let connection = servers.get(&tool_info.server_id).ok_or_else(|| {
260                McpError::new(format!("Server not found: {}", tool_info.server_id))
261            })?;
262
263            // Extract original tool name (remove prefix if present)
264            let original_name = if let Some(prefix) = &connection.config.tool_prefix {
265                let prefix_with_underscore = format!("{}_", prefix);
266                if tool_name.starts_with(&prefix_with_underscore) {
267                    tool_name[prefix_with_underscore.len()..].to_string()
268                } else {
269                    tool_name.to_string()
270                }
271            } else {
272                tool_name.to_string()
273            };
274
275            (original_name, connection.config.clone())
276        };
277
278        // Acquire concurrency permit
279        let _permit = self
280            .concurrency_semaphore
281            .acquire()
282            .await
283            .map_err(|_| McpError::new("Failed to acquire concurrency permit"))?;
284
285        // Execute with timeout
286        let timeout_duration = Duration::from_secs(self.config.tool_timeout_secs.unwrap_or(30));
287
288        let result = tokio::time::timeout(
289            timeout_duration,
290            self.execute_tool(&server_config, &original_name, arguments),
291        )
292        .await
293        .map_err(|_| {
294            McpError::new(format!(
295                "Tool call timed out after {} seconds",
296                timeout_duration.as_secs()
297            ))
298        })??;
299
300        Ok(result)
301    }

303    /// Execute a tool on a specific server
304    async fn execute_tool(
305        &self,
306        config: &McpServerConfig,
307        tool_name: &str,
308        arguments: serde_json::Value,
309    ) -> Result<ToolResult> {
310        let params = arguments.as_object().cloned().unwrap_or_default();
311
312        let result = match &config.source {
313            McpServerSource::Http { url, .. } => {
314                mcp_methods::run_tool_via_http(url.clone(), tool_name.to_string(), params).await?
315            }
316            McpServerSource::Sse { url, .. } => {
317                mcp_methods::run_tool_via_sse(url.clone(), tool_name.to_string(), params).await?
318            }
319            McpServerSource::Process {
320                command, args, env, ..
321            } => {
322                let cmd_str = if args.is_empty() {
323                    command.clone()
324                } else {
325                    format!("{} {}", command, args.join(" "))
326                };
327                mcp_methods::run_tool_via_command(
328                    cmd_str,
329                    tool_name.to_string(),
330                    env.clone().unwrap_or_default(),
331                    params,
332                )
333                .await?
334            }
335            McpServerSource::WebSocket { .. } => {
336                return Err(McpError::new("WebSocket transport not yet implemented"));
337            }
338        };
339
340        // Convert CallToolResult to ToolResult
341        let content: Vec<hanzo_mcp_core::ContentBlock> = result
342            .content
343            .iter()
344            .filter_map(|c| {
345                c.as_text().map(|t| hanzo_mcp_core::ContentBlock::Text {
346                    text: t.text.clone(),
347                })
348            })
349            .collect();
350
351        Ok(ToolResult {
352            content,
353            is_error: result.is_error.unwrap_or(false),
354        })
355    }

381    /// List all resources from all connected servers
382    pub async fn list_resources(&self) -> Result<Vec<McpResourceInfo>> {
383        let servers = self.servers.read().await;
384        let mut all_resources = Vec::new();
385
386        for (server_id, connection) in servers.iter() {
387            match self.list_server_resources(&connection.config).await {
388                Ok(resources) => {
389                    for resource in resources {
390                        all_resources.push(McpResourceInfo {
391                            uri: resource.uri.to_string(),
392                            name: resource.name.to_string(),
393                            description: resource.description.as_ref().map(|d| d.to_string()),
394                            mime_type: resource.mime_type.as_ref().map(|m| m.to_string()),
395                            server_id: server_id.clone(),
396                            server_name: connection.config.name.clone(),
397                        });
398                    }
399                }
400                Err(e) => {
401                    warn!("Failed to list resources from server {}: {}", server_id, e);
402                }
403            }
404        }
405
406        Ok(all_resources)
407    }

438    /// Read a resource by URI
439    ///
440    /// The URI must match a resource from one of the connected servers.
441    pub async fn read_resource(&self, uri: &str) -> Result<String> {
442        // First, find which server has this resource
443        let resources = self.list_resources().await?;
444        let resource_info = resources
445            .iter()
446            .find(|r| r.uri == uri)
447            .ok_or_else(|| McpError::new(format!("Resource not found: {}", uri)))?;
448
449        let servers = self.servers.read().await;
450        let connection = servers.get(&resource_info.server_id).ok_or_else(|| {
451            McpError::new(format!("Server not found: {}", resource_info.server_id))
452        })?;
453
454        self.read_server_resource(&connection.config, uri).await
455    }

460            McpServerSource::Http { url, .. } => {
461                mcp_methods::read_resource_via_http(url, uri).await
462            }
463            McpServerSource::Sse { url, .. } => mcp_methods::read_resource_via_sse(url, uri).await,
464            McpServerSource::Process {
465                command, args, env, ..
466            } => {
467                let cmd_str = if args.is_empty() {
468                    command.clone()
469                } else {
470                    format!("{} {}", command, args.join(" "))
471                };
472                mcp_methods::read_resource_via_command(&cmd_str, uri, env.clone()).await
473            }
474            McpServerSource::WebSocket { .. } => Err(McpError::new(
475                "WebSocket transport not yet supported for resources",
476            )),
477        }
478    }
