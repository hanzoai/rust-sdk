14pub async fn list_tools_via_command(
15    cmd_str: &str,
16    config: Option<HashMap<String, String>>,
17) -> Result<Vec<Tool>> {
18    let (env_vars, cmd_executable, cmd_args) = disect_command(cmd_str.to_string());
19    let (adapted_program, adapted_args, adapted_envs) =
20        CommandWrappedInShellBuilder::wrap_in_shell_as_values(
21            cmd_executable,
22            Some(cmd_args),
23            Some(env_vars),
24        );
25    let mut cmd = Command::new(adapted_program);
26    cmd.kill_on_drop(true);
27    cmd.envs(adapted_envs);
28    cmd.envs(config.unwrap_or_default());
29    cmd.args(adapted_args);
30
31    // Retain the TokioChildProcess so we can wait on it after cancellation
32    let child_process = TokioChildProcess::new(cmd).map_err(|e| McpError {
33        message: format!("{}", e),
34    })?;
35    let service = ().serve(child_process).await.map_err(|e| McpError {
36        message: format!("{}", e),
37    })?;
38    // 2. Initialize the MCP server
39    service.peer_info();
40
41    // 3. Call the standard MCP `list_tools` method
42    let tools = service
43        .list_all_tools()
44        .await
45        .inspect_err(|e| log::error!("error listing tools: {:?}", e));
46
47    // 4. Gracefully shut down the service (drops stdio, child should exit)
48    let _ = service
49        .cancel()
50        .await
51        .inspect_err(|e| log::error!("error cancelling sse service: {:?}", e));
52
53    Ok(tools.unwrap())
54}

416/// Read a resource from an MCP server via HTTP
417pub async fn read_resource_via_http(url: &str, uri: &str) -> Result<String> {
418    let transport = StreamableHttpClientTransport::from_uri(url);
419    let client_info = ClientInfo {
420        protocol_version: Default::default(),
421        capabilities: ClientCapabilities::default(),
422        client_info: Implementation {
423            name: "hanzo_mcp_http_client".to_string(),
424            version: env!("CARGO_PKG_VERSION").to_string(),
425            title: None,
426            icons: None,
427            website_url: None,
428        },
429    };
430    let client = client_info.serve(transport).await.map_err(|e| McpError {
431        message: format!("HTTP client connection error: {:?}", e),
432    })?;
433
434    let _ = client.peer_info();
435
436    let result = client
437        .read_resource(ReadResourceRequestParam {
438            uri: uri.to_string().into(),
439        })
440        .await
441        .inspect_err(|e| log::error!("error reading resource: {:?}", e));
442
443    let _ = client
444        .cancel()
445        .await
446        .inspect_err(|e| log::error!("error cancelling http service: {:?}", e));
447
448    let read_result = result.map_err(|e| McpError {
449        message: format!("{}", e),
450    })?;
451
452    // Extract text content from the result
453    let text = read_result
454        .contents
455        .iter()
456        .filter_map(|c| match c {
457            rmcp::model::ResourceContents::TextResourceContents { text, .. } => {
458                Some(text.to_string())
459            }
460            _ => None,
461        })
462        .collect::<Vec<_>>()
463        .join("\n");
464
465    Ok(text)
466}
