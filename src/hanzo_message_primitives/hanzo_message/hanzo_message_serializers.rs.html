1use super::hanzo_message::HanzoVersion;

3use serde::{Deserialize, Deserializer, Serialize, Serializer};
4
5impl Serialize for HanzoVersion {
6    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
7    where
8        S: Serializer,
9    {
10        let version = match *self {
11            HanzoVersion::V1_0 => "V1_0",
12            HanzoVersion::Unsupported => "Unsupported",
13        };
14        serializer.serialize_str(version)
15    }
16}
17
18impl<'de> Deserialize<'de> for HanzoVersion {
19    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
20    where
21        D: Deserializer<'de>,
22    {
23        let version = String::deserialize(deserializer)?;
24        Ok(match version.as_str() {
25            "V1_0" => HanzoVersion::V1_0,
26            _ => HanzoVersion::Unsupported,
27        })
28    }
29}
