16impl HanzoMessageBuilder {
17    #[allow(clippy::too_many_arguments)]
18    #[allow(dead_code)]
19    pub fn create_generic_invoice_message(
20        payload: impl Serialize,
21        schema_type: MessageSchemaType,
22        my_encryption_secret_key: EncryptionStaticKey,
23        my_signature_secret_key: SigningKey,
24        receiver_public_key: EncryptionPublicKey,
25        sender: HanzoNameString,
26        sender_subidentity: HanzoNameString,
27        node_receiver: HanzoNameString,
28        node_receiver_subidentity: HanzoNameString,
29        external_metadata: Option<ExternalMetadata>,
30    ) -> Result<HanzoMessage, &'static str> {
31        let body = serde_json::to_string(&payload)
32            .map_err(|_| "Failed to serialize job creation to JSON")?;
33
34        // Convert the encryption secret key to a public key and print it
35        let my_encryption_public_key = EncryptionPublicKey::from(&my_encryption_secret_key);
36        let my_enc_string = encryption_public_key_to_string(my_encryption_public_key);
37
38        let mut my_enc_string = my_enc_string;
39        let mut sender_subidentity = sender_subidentity;
40        if let Some(external_metadata) = external_metadata {
41            if !external_metadata.other.is_empty() && !external_metadata.intra_sender.is_empty() {
42                my_enc_string = external_metadata.other;
43                sender_subidentity = external_metadata.intra_sender.clone();
44            }
45        }
46
47        HanzoMessageBuilder::new(
48            my_encryption_secret_key,
49            my_signature_secret_key,
50            receiver_public_key,
51        )
52        .message_raw_content(body)
53        .internal_metadata_with_schema(
54            sender_subidentity.clone(),
55            node_receiver_subidentity.clone(),
56            "".to_string(),
57            schema_type,
58            EncryptionMethod::None,
59            None,
60        )
61        .body_encryption(EncryptionMethod::DiffieHellmanChaChaPoly1305)
62        .external_metadata_with_other_and_intra_sender(
63            node_receiver,
64            sender,
65            my_enc_string,
66            sender_subidentity,
67        )
68        .build()
69    }
