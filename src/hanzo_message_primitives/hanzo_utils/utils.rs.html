4/// Cleans an input string to ensure that it does not have any
5/// characters which would break a VRPath, or cause issues generally for the VectorFS.
6pub fn clean_string(s: &str) -> String {
7    s.replace("/", "-").replace(":", "_")
8}

10/// Hashes a String using Blake3, returning the hash as an output String
11pub fn hash_string(input: &str) -> String {
12    let mut hasher = blake3::Hasher::new();
13    hasher.update(input.as_bytes());
14    let result = hasher.finalize();
15    hex::encode(result.as_bytes())
16}

18/// Generates a random hex String
19pub fn random_string() -> String {
20    let mut key = [0u8; 32];
21    rand::rngs::OsRng.fill_bytes(&mut key);
22
23    let mut hasher = Hasher::new();
24    hasher.update(&key);
25    let hash = hasher.finalize();
26
27    hex::encode(hash.as_bytes())
28}
