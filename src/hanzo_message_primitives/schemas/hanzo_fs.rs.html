6/// Represents a file that has been parsed and indexed (e.g., split into chunks and possibly embedded).
7/// This record stores metadata about the parsing process and the file itself, including its relative
8/// path, extension, descriptions, and token/character counts.
9#[derive(Debug, Clone, Serialize, Deserialize)]
10pub struct ParsedFile {
11    /// Unique identifier for the parsed file entry.
12    pub id: Option<i64>,
13    /// The file's path relative to some base directory (e.g., "docs/manual.txt").
14    pub relative_path: String,
15    /// The original file extension (e.g., "txt", "md", "pdf").
16    pub original_extension: Option<String>,
17    /// A human-readable description of the file.
18    pub description: Option<String>,
19    /// The source of the file content (e.g., a URL or system component).
20    pub source: Option<String>,
21    /// The name or type of the embedding model used if embeddings were generated.
22    pub embedding_model_used: Option<String>,
23    /// Keywords or tags derived from or associated with the file.
24    pub keywords: Option<String>,
25    /// Information about how the file is distributed or shared.
26    pub distribution_info: Option<String>,
27    /// The timestamp when the file was parsed or created (in a UNIX timestamp format).
28    pub created_time: Option<i64>,
29    /// Arbitrary tags associated with the file for categorization or filtering.
30    pub tags: Option<String>,
31    /// The total number of tokens in the file (if known).
32    pub total_tokens: Option<i64>,
33    /// The total number of characters in the file (if known).
34    pub total_characters: Option<i64>,
35}

37/// Represents a chunk of a processed file.
38#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq, Hash)]
39pub struct HanzoFileChunk {
40    /// Unique identifier for the file chunk.
41    pub chunk_id: Option<i64>,
42    /// Identifier of the parsed file this chunk is associated with.
43    pub parsed_file_id: i64,
44    /// Sequence number of the chunk, indicating its order within the file.
45    pub position: i64,
46    /// The text content of this particular chunk.
47    pub content: String,
48}
49
50/// Represents an embedding of a file chunk.
51#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
52pub struct HanzoFileChunkEmbedding {
53    /// Identifier of the file chunk this embedding is associated with.
54    pub chunk_id: i64,
55    /// Embedding vector for the file chunk.
56    pub embedding: Vec<f32>,
57}
58
59/// A struct that holds a collection of `HanzoFileChunk`.
60#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
61pub struct HanzoFileChunkCollection {
62    /// A set of chunks related to a parsed file.
63    pub chunks: Vec<HanzoFileChunk>,
64    /// A map of parsed file IDs to their associated paths.
65    pub paths: Option<HashMap<i64, HanzoPath>>,
66}
67
68impl HanzoFileChunkCollection {
69    /// Checks if the collection of chunks is empty.
70    pub fn is_empty(&self) -> bool {
71        self.chunks.is_empty()
72    }
