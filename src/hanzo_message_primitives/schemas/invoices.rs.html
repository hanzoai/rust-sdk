9/// Generates a random 32-byte nonce encoded as a hex string prefixed with `0x`.
10/// This mimics the nonce used by x402 payment requests and is used as the
11/// unique identifier for invoices.
12pub fn generate_x402_nonce() -> String {
13    let mut bytes = [0u8; 32];
14    rand::thread_rng().fill_bytes(&mut bytes);
15    format!("0x{}", hex::encode(bytes))
16}

77impl fmt::Display for InvoiceError {
78    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
79        match self {
80            InvoiceError::InvalidToolKeyFormat => write!(f, "Invalid tool_key_name format"),
81            InvoiceError::NodeNameMismatch { expected, found } => {
82                write!(
83                    f,
84                    "Node name mismatch: expected {}, found {}",
85                    expected, found
86                )
87            }
88            InvoiceError::OperationFailed(msg) => write!(f, "Operation failed: {}", msg),
89        }
90    }
91}

108impl InvoiceRequest {
109    pub fn validate_and_convert_tool_key(
110        &self,
111        node_name: &HanzoName,
112    ) -> Result<String, InvoiceError> {
113        // Extract the node name from the tool_key_name
114        let parts: Vec<&str> = self.tool_key_name.split(":::").collect();
115        if parts.len() < 3 {
116            return Err(InvoiceError::OperationFailed(
117                "Invalid tool_key_name format".to_string(),
118            ));
119        }
120
121        let node_name_part = parts[0];
122        let author = parts[1];
123        let tool_name = parts[2];
124
125        // Normalize both node_name_part and node_name for comparison
126        let normalized_node_name_part = ToolRouterKey::sanitize(node_name_part);
127        let normalized_node_name = ToolRouterKey::sanitize(&node_name.to_string());
128
129        // Validate that the normalized node name part matches our normalized node_name
130        if normalized_node_name_part != normalized_node_name {
131            return Err(InvoiceError::OperationFailed(format!(
132                "Node name in tool_key_name does not match our node_name (expected: {}, found: {})",
133                normalized_node_name, normalized_node_name_part
134            )));
135        }
136
137        // Convert the tool_key_name to the actual tool_key_name
138        let actual_tool_key_name = format!("local:::{}:::{}", author, tool_name);
139
140        Ok(actual_tool_key_name)
141    }
142}
