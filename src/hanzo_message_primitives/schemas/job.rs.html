24#[derive(Clone, Debug)]
25pub struct Job {
26    /// Based on uuid
27    pub job_id: String,
28    /// Marks if the job is hidden or not. Hidden jobs are not shown in the UI to avoid spamming
29    pub is_hidden: bool,
30    /// Format: "20230702T20533481346" or Utc::now().format("%Y%m%dT%H%M%S%f").to_string();
31    pub datetime_created: String,
32    /// Marks if the job is finished or not
33    pub is_finished: bool,
34    /// Identity of the parent agent. We just use a full identity name for simplicity
35    pub parent_agent_or_llm_provider_id: String,
36    /// (Simplified version) What VectorResources the Job has access to when performing vector searches
37    pub scope: MinimalJobScope,
38    // /// (Full version) What VectorResources the Job has access to when performing vector searches, including files
39    // pub scope_with_files: Option<MinimalJobScope>,
40    /// A list of messages that were generated as output from the latest Job step
41    pub step_history: Vec<HanzoMessage>,
42    /// An inbox where messages to the agent from the user and messages from the agent are stored,
43    /// enabling each job to have a classical chat/conversation UI
44    pub conversation_inbox_name: InboxName,
45    /// A link to the UI where the user can view the job e.g. Sheet UI
46    pub associated_ui: Option<AssociatedUI>,
47    /// The job's configuration
48    pub config: Option<JobConfig>,
49    /// Keep track of forked jobs
50    pub forked_jobs: Vec<ForkedJob>,
51}
