7/// The parameters of the function.
8#[derive(Debug, Default, Clone, PartialEq, Eq, Serialize, Deserialize)]
9pub struct FunctionParameters {
10    /// The type of the parameters.
11    #[serde(rename = "type")]
12    pub type_: String,
13    /// The properties of the parameters.
14    pub properties: Value,
15    /// The required parameters.
16    pub required: Vec<String>,
17}

19/// The details of the function to be called.
20#[derive(Debug, Default, Clone, PartialEq, Eq, Serialize, Deserialize)]
21pub struct FunctionDetails {
22    /// The name of the function.
23    pub name: String,
24    /// The tool router key of the function.
25    #[serde(skip_serializing_if = "Option::is_none")]
26    pub tool_router_key: Option<String>,
27    /// The description of the function.
28    pub description: String,
29    /// The parameters of the function.
30    pub parameters: FunctionParameters,
31}

65/// The message structure for LLM communication.
66#[derive(Default, Clone, PartialEq, Eq, Serialize, Deserialize)]
67pub struct LlmMessage {
68    /// The role of the message's author. One of `system`, `user`, `assistant`, or `function`.
69    #[serde(skip_serializing_if = "Option::is_none")]
70    pub role: Option<String>,
71    /// The contents of the message.
72    /// `content` is required for all messages except assistant messages with function calls.
73    #[serde(skip_serializing_if = "Option::is_none")]
74    pub content: Option<String>,
75    /// The name of the author of this message. `name` is required if role is function,
76    /// and it should be the name of the function whose response is in the `content`.
77    /// May contain a-z, A-Z, 0-9, and underscores, with a maximum length of 64 characters.
78    #[serde(skip_serializing_if = "Option::is_none")]
79    pub name: Option<String>,
80    /// The detailed function call structure.
81    #[serde(skip_serializing_if = "Option::is_none")]
82    pub function_call: Option<DetailedFunctionCall>,
83    /// The available functions.
84    #[serde(skip_serializing_if = "Option::is_none")]
85    pub functions: Option<Vec<FunctionDetails>>,
86    /// The images associated with the message.
87    #[serde(skip_serializing_if = "Option::is_none")]
88    pub images: Option<Vec<String>>,
89    /// The videos associated with the message.
90    #[serde(skip_serializing_if = "Option::is_none")]
91    pub videos: Option<Vec<String>>,
92    /// The audios associated with the message.
93    #[serde(skip_serializing_if = "Option::is_none")]
94    pub audios: Option<Vec<String>>,
95    /// The tool calls associated with the message.
96    #[serde(skip_serializing_if = "Option::is_none")]
97    pub tool_calls: Option<Vec<ToolCall>>,
98}

100impl fmt::Debug for LlmMessage {
101    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
102        f.debug_struct("LlmMessage")
103            .field("role", &self.role)
104            .field("content", &self.content)
105            .field("name", &self.name)
106            .field("function_call", &self.function_call)
107            .field("functions", &self.functions)
108            .field(
109                "images",
110                &self.images.as_ref().map(|images| {
111                    images
112                        .iter()
113                        .map(|img| {
114                            if img.len() > 20 {
115                                format!("{}...", &img[..20])
116                            } else {
117                                img.clone()
118                            }
119                        })
120                        .collect::<Vec<String>>()
121                }),
122            )
123            .field(
124                "videos",
125                &self.videos.as_ref().map(|videos| {
126                    videos
127                        .iter()
128                        .map(|vid| {
129                            if vid.len() > 20 {
130                                format!("{}...", &vid[..20])
131                            } else {
132                                vid.clone()
133                            }
134                        })
135                        .collect::<Vec<String>>()
136                }),
137            )
138            .field(
139                "audios",
140                &self.audios.as_ref().map(|audios| {
141                    audios
142                        .iter()
143                        .map(|aud| {
144                            if aud.len() > 20 {
145                                format!("{}...", &aud[..20])
146                            } else {
147                                aud.clone()
148                            }
149                        })
150                        .collect::<Vec<String>>()
151                }),
152            )
153            .field("tool_calls", &self.tool_calls)
154            .finish()
155    }

158#[derive(Debug, Error)]
159pub enum LlmMessageError {
160    #[error("Failed to parse JSON: {0}")]
161    JsonParseError(#[from] SerdeError),
162}
163
164impl LlmMessage {
165    /// Imports an LlmMessage from a JSON value.
166    pub fn import_functions_from_value(value: Value) -> Result<Self, LlmMessageError> {
167        let role = None;
168        let content = None;
169        let name = None;
170
171        let images = value.get("images").and_then(|v| {
172            v.as_array().map(|arr| {
173                arr.iter()
174                    .filter_map(|v| v.as_str().map(String::from))
175                    .collect()
176            })
177        });
178
179        let videos = value.get("videos").and_then(|v| {
180            v.as_array().map(|arr| {
181                arr.iter()
182                    .filter_map(|v| v.as_str().map(String::from))
183                    .collect()
184            })
185        });
186
187        let audios = value.get("audios").and_then(|v| {
188            v.as_array().map(|arr| {
189                arr.iter()
190                    .filter_map(|v| v.as_str().map(String::from))
191                    .collect::<Vec<String>>()
192            })
193        });
194
195        // Extract the functions from the "function" key
196        let functions_value = value.get("function").ok_or_else(|| {
197            let err_msg = "Missing 'function' key in JSON";
198            eprintln!("{}", err_msg);
199            LlmMessageError::JsonParseError(serde::de::Error::custom(err_msg))
200        })?;
201
202        let functions: Vec<FunctionDetails> = match functions_value {
203            Value::Array(arr) => serde_json::from_value(Value::Array(arr.to_vec()))?,
204            Value::Object(_) => vec![serde_json::from_value(functions_value.clone())?],
205            _ => {
206                return Err(LlmMessageError::JsonParseError(serde::de::Error::custom(
207                    "Invalid JSON format for functions",
208                )))
209            }
210        };
211
212        Ok(LlmMessage {
213            role,
214            content,
215            name,
216            function_call: None,
217            functions: Some(functions),
218            images,
219            videos,
220            audios,
221            tool_calls: None,
222        })
223    }
224}

231    #[test]
232    fn test_import_functions_from_value_object() {
233        let json_value = json!({
234            "function": {
235                "name": "concat_strings",
236                "description": "Concatenates 2 to 4 strings.",
237                "parameters": {
238                    "type": "object",
239                    "properties": {
240                        "first_string": {
241                            "type": "string",
242                            "description": "The first string to concatenate"
243                        },
244                        "second_string": {
245                            "type": "string",
246                            "description": "The second string to concatenate"
247                        },
248                        "third_string": {
249                            "type": "string",
250                            "description": "The third string to concatenate (optional)"
251                        },
252                        "fourth_string": {
253                            "type": "string",
254                            "description": "The fourth string to concatenate (optional)"
255                        }
256                    },
257                    "required": ["first_string", "second_string"]
258                }
259            },
260            "type": "function"
261        });
262
263        let message = LlmMessage::import_functions_from_value(json_value)
264            .expect("Failed to import functions from value");
265
266        assert!(message.role.is_none());
267        assert!(message.content.is_none());
268        assert!(message.name.is_none());
269        assert!(message.function_call.is_none());
270        assert!(message.functions.is_some());
271
272        let functions = message.functions.unwrap();
273        assert_eq!(functions.len(), 1);
274
275        let function = &functions[0];
276        assert_eq!(function.name, "concat_strings");
277        assert_eq!(function.description, "Concatenates 2 to 4 strings.");
278        assert_eq!(function.parameters.type_, "object");
279
280        let properties = function.parameters.properties.as_object().unwrap();
281        assert_eq!(
282            properties.get("first_string").unwrap().get("type").unwrap(),
283            "string"
284        );
285        assert_eq!(
286            properties
287                .get("first_string")
288                .unwrap()
289                .get("description")
290                .unwrap(),
291            "The first string to concatenate"
292        );
293        assert_eq!(
294            properties
295                .get("second_string")
296                .unwrap()
297                .get("type")
298                .unwrap(),
299            "string"
300        );
301        assert_eq!(
302            properties
303                .get("second_string")
304                .unwrap()
305                .get("description")
306                .unwrap(),
307            "The second string to concatenate"
308        );
309        assert_eq!(
310            properties.get("third_string").unwrap().get("type").unwrap(),
311            "string"
312        );
313        assert_eq!(
314            properties
315                .get("third_string")
316                .unwrap()
317                .get("description")
318                .unwrap(),
319            "The third string to concatenate (optional)"
320        );
321        assert_eq!(
322            properties
323                .get("fourth_string")
324                .unwrap()
325                .get("type")
326                .unwrap(),
327            "string"
328        );
329        assert_eq!(
330            properties
331                .get("fourth_string")
332                .unwrap()
333                .get("description")
334                .unwrap(),
335            "The fourth string to concatenate (optional)"
336        );
337
338        let required = &function.parameters.required;
339        assert_eq!(required.len(), 2);
340        assert!(required.contains(&"first_string".to_string()));
341        assert!(required.contains(&"second_string".to_string()));
342    }
