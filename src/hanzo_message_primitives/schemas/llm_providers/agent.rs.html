14#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, ToSchema)]
15#[serde(rename_all = "snake_case")]
16pub struct Agent {
17    pub name: String,
18    pub agent_id: String,
19    pub full_identity_name: HanzoName,
20    pub llm_provider_id: String,
21    pub ui_description: String,
22    pub knowledge: Vec<String>,
23    pub storage_path: String,
24    #[serde(
25        serialize_with = "ToolRouterKey::serialize_tool_router_keys",
26        deserialize_with = "ToolRouterKey::deserialize_tool_router_keys"
27    )]
28    pub tools: Vec<ToolRouterKey>,
29    pub debug_mode: bool,
30    pub config: Option<JobConfig>,
31    #[serde(default)]
32    pub scope: MinimalJobScope,
33    pub cron_tasks: Option<Vec<CronTask>>,
34    /// Tool configuration overrides keyed by tool ID with a map of configuration values
35    #[serde(default)]
36    pub tools_config_override: Option<HashMap<String, HashMap<String, Value>>>,
37    #[serde(default)]
38    pub edited: bool,
39}
40
41impl Agent {
42    pub fn sanitize_config(&mut self) {
43        self.tools_config_override = None;
44    }
45}

52    #[test]
53    fn test_agent_deserialize_with_valid_tools() {
54        let json_data = json!({
55            "name": "test_agent",
56            "agent_id": "test123",
57            "full_identity_name": "test.agent",
58            "llm_provider_id": "test_provider",
59            "ui_description": "Test Agent",
60            "knowledge": ["test knowledge"],
61            "storage_path": "/test/path",
62            "tools": [
63                "local:::__some_name:::tool1",
64                "local:::__some_name:::tool2:::1.0"
65            ],
66            "debug_mode": false,
67            "config": null
68        });
69
70        let agent: Agent = serde_json::from_value(json_data).unwrap();
71        assert_eq!(agent.tools.len(), 2);
72        assert_eq!(agent.tools[0].source, "local");
73        assert_eq!(agent.tools[0].author, "__some_name");
74        assert_eq!(agent.tools[0].name, "tool1");
75        assert_eq!(agent.tools[0].version, None);
76        assert_eq!(agent.tools[1].version, Some("1.0".to_string()));

119    #[test]
120    fn test_agent_with_tools_config_override() {
121        use serde_json::json;
122        use std::collections::HashMap;
123
124        // Create a test agent with tools_config_override
125        let mut tools_config = HashMap::new();
126        let mut config_values = HashMap::new();
127        config_values.insert("api_key".to_string(), json!("test-api-key"));
128        config_values.insert("timeout".to_string(), json!(30));
129        tools_config.insert("local:::__some_name:::tool1".to_string(), config_values);
130
131        let json_data = json!({
132            "name": "test_agent",
133            "agent_id": "test123",
134            "full_identity_name": "test.agent",
135            "llm_provider_id": "test_provider",
136            "ui_description": "Test Agent",
137            "knowledge": ["test knowledge"],
138            "storage_path": "/test/path",
139            "tools": [
140                "local:::__some_name:::tool1",
141                "local:::__some_name:::tool2:::1.0"
142            ],
143            "debug_mode": false,
144            "config": null,
145            "tools_config_override": tools_config
146        });
147
148        let agent: Agent = serde_json::from_value(json_data).unwrap();
149
150        // Verify tools_config_override was correctly deserialized
151        assert!(agent.tools_config_override.is_some());
152        let config_map = agent.tools_config_override.unwrap();
153        assert!(config_map.contains_key("local:::__some_name:::tool1"));
154        let tool_config = config_map.get("local:::__some_name:::tool1").unwrap();
155        assert_eq!(
156            tool_config.get("api_key").unwrap().as_str().unwrap(),
157            "test-api-key"
158        );
159        assert_eq!(tool_config.get("timeout").unwrap().as_i64().unwrap(), 30);
160    }
