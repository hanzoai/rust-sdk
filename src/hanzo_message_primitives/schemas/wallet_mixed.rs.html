13/// Represents an address in a wallet.
14#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, ToSchema)]
15pub struct Address {
16    /// The ID of the wallet that owns the address.
17    pub wallet_id: String,
18    /// The ID of the blockchain network.
19    pub network_id: x402_types::Network,
20    /// The public key from which the address is derived.
21    pub public_key: Option<String>,
22    /// The onchain address derived on the server-side.
23    pub address_id: String,
24}
25
26impl From<Address> for PublicAddress {
27    fn from(address: Address) -> Self {
28        PublicAddress {
29            network_id: address.network_id,
30            address_id: address.address_id,
31        }
32    }
33}

71/// Represents an asset onchain scoped to a particular network.
72#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize, ToSchema)]
73pub struct Asset {
74    /// The ID of the blockchain network.
75    pub network_id: x402_types::Network,
76    /// The ID for the asset on the network.
77    pub asset_id: String,
78    /// The number of decimals the asset supports. This is used to convert from atomic units to base units.
79    pub decimals: Option<u32>,
80    /// The optional contract address for the asset. This will be specified for smart contract-based assets, for
81    /// example ERC20s.
82    pub contract_address: Option<String>,
83}
84
85#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
86pub enum AssetType {
87    ETH,
88    USDC,
89    KAI,
90}

92impl Asset {
93    pub fn new(asset_type: AssetType, network: &x402_types::Network) -> Option<Self> {
94        match (asset_type, network) {
95            (AssetType::ETH, _) => Some(Asset {
96                network_id: network.clone(),
97                asset_id: "ETH".to_string(),
98                decimals: Some(18),
99                contract_address: None,
100            }),
101            (AssetType::USDC, x402_types::Network::BaseSepolia) => Some(Asset {
102                network_id: x402_types::Network::BaseSepolia,
103                asset_id: "USDC".to_string(),
104                decimals: Some(6),
105                contract_address: Some("0x036CbD53842c5426634e7929541eC2318f3dCF7e".to_string()),
106            }),
107            (AssetType::USDC, x402_types::Network::Base) => Some(Asset {
108                network_id: x402_types::Network::Base,
109                asset_id: "USDC".to_string(),
110                decimals: Some(6),
111                contract_address: Some("0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913".to_string()),
112            }),
113            (AssetType::KAI, _) => None, // KAI is not specified for any network yet
114            (AssetType::USDC, x402_types::Network::AvalancheFuji) => None, // USDC not specified for Avalanche Fuji
115            (AssetType::USDC, x402_types::Network::Avalanche) => None, // USDC not specified for Avalanche
116        }
117    }

311impl fmt::Display for NetworkIdentifier {
312    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
313        let display_str = match self {
314            NetworkIdentifier::BaseSepolia => "Base Sepolia",
315            NetworkIdentifier::BaseMainnet => "Base Mainnet",
316            NetworkIdentifier::EthereumSepolia => "Ethereum Sepolia",
317            NetworkIdentifier::EthereumMainnet => "Ethereum Mainnet",
318            NetworkIdentifier::ArbitrumSepolia => "Arbitrum Sepolia",
319            NetworkIdentifier::ArbitrumMainnet => "Arbitrum Mainnet",
320            NetworkIdentifier::Anvil => "Anvil",
321        };
322        write!(f, "{}", display_str)
323    }
324}

326/// Represents an onchain transaction.
327#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
328pub struct Transaction {
329    /// The ID of the blockchain network.
330    pub network_id: String,
331    /// The onchain address of the sender.
332    pub from_address_id: String,
333    /// The onchain address of the recipient.
334    pub to_address_id: Option<String>,
335    /// The unsigned payload of the transaction. This is the payload that needs to be signed by the sender.
336    pub unsigned_payload: String,
337    /// The signed payload of the transaction. This is the payload that has been signed by the sender.
338    pub signed_payload: Option<String>,
339    /// The hash of the transaction.
340    pub transaction_hash: Option<String>,
341    /// The link to view the transaction on a block explorer. This is optional and may not be present for all
342    /// transactions.
343    pub transaction_link: Option<String>,
344    /// The status of the transaction.
345    pub status: TransactionStatusEnum,
346}
347
348/// Enum representing the status of the transaction.
349#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
350pub enum TransactionStatusEnum {
351    Pending,
352    Signed,
353    Broadcast,
354    Complete,
355    Failed,
356}
357
358/// Enum representing the type of transaction.
359#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
360pub enum TransactionType {
361    Transfer,
362}

364/// Represents a transfer of an asset from one address to another.
365#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
366pub struct Transfer {
367    /// The ID of the blockchain network.
368    network_id: String,
369    /// The ID of the wallet that owns the from address.
370    wallet_id: String,
371    /// The onchain address of the sender.
372    address_id: String,
373    /// The onchain address of the recipient.
374    destination: String,
375    /// The amount in the atomic units of the asset.
376    amount: String,
377    /// The ID of the asset being transferred.
378    asset_id: String,
379    /// The asset associated with the transfer.
380    asset: Asset,
381    /// The ID of the transfer.
382    transfer_id: String,
383    /// The transaction associated with the transfer.
384    transaction: Option<Transaction>,
385    /// The sponsored send associated with the transfer.
386    // sponsored_send: Option<SponsoredSend>,
387    /// The unsigned payload of the transfer. This is the payload that needs to be signed by the sender.
388    unsigned_payload: Option<String>,
389    /// The signed payload of the transfer. This is the payload that has been signed by the sender.
390    signed_payload: Option<String>,
391    /// The hash of the transfer transaction.
392    transaction_hash: Option<String>,
393    /// The status of the transfer.
394    status: Option<TransferStatusEnum>,
395    /// Whether the transfer uses sponsored gas.
396    gasless: bool,
397}
