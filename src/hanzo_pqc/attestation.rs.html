6/// TEE attestation verifier trait
7pub trait AttestationVerifier: Send + Sync {
8    /// Verify CPU TEE attestation
9    fn verify_cpu_attestation(&self, quote: &[u8], expected_measurement: &[u8]) -> Result<bool>;
10
11    /// Verify GPU attestation via NRAS
12    fn verify_gpu_attestation(&self, nras_token: &[u8]) -> Result<GpuAttestationResult>;
13}

69/// SEV-SNP attestation report parser
70#[cfg(feature = "cpu-tee")]
71pub mod sev_snp {
72    use super::*;
73
74    #[derive(Debug, Clone, Serialize, Deserialize)]
75    pub struct SevSnpReport {
76        pub version: u32,
77        pub guest_svn: u32,
78        pub policy: u64,
79        pub family_id: [u8; 16],
80        pub image_id: [u8; 16],
81        pub vmpl: u32,
82        pub signature_algo: u32,
83        pub platform_version: u64,
84        pub platform_info: u64,
85        pub author_key_en: u32,
86        pub report_data: [u8; 64],
87        pub measurement: [u8; 48],
88        pub host_data: [u8; 32],
89        pub id_key_digest: [u8; 48],
90        pub author_key_digest: [u8; 48],
91        pub report_id: [u8; 32],
92        pub report_id_ma: [u8; 32],
93        pub reported_tcb: u64,
94        pub chip_id: [u8; 64],
95        pub signature: [u8; 512],
96    }
97
98    impl SevSnpReport {
99        pub fn parse(data: &[u8]) -> Result<Self> {
100            if data.len() < 1184 {
101                return Err(PqcError::AttestationError(
102                    "Invalid SEV-SNP report size".into(),
103                ));
104            }
105
106            // Parse binary format (simplified)
107            Ok(Self {
108                version: u32::from_le_bytes([data[0], data[1], data[2], data[3]]),
109                guest_svn: u32::from_le_bytes([data[4], data[5], data[6], data[7]]),
110                policy: u64::from_le_bytes(data[8..16].try_into().unwrap()),
111                family_id: data[16..32].try_into().unwrap(),
112                image_id: data[32..48].try_into().unwrap(),
113                vmpl: u32::from_le_bytes([data[48], data[49], data[50], data[51]]),
114                signature_algo: u32::from_le_bytes([data[52], data[53], data[54], data[55]]),
115                platform_version: u64::from_le_bytes(data[56..64].try_into().unwrap()),
116                platform_info: u64::from_le_bytes(data[64..72].try_into().unwrap()),
117                author_key_en: u32::from_le_bytes([data[72], data[73], data[74], data[75]]),
118                report_data: data[76..140].try_into().unwrap(),
119                measurement: data[140..188].try_into().unwrap(),
120                host_data: data[188..220].try_into().unwrap(),
121                id_key_digest: data[220..268].try_into().unwrap(),
122                author_key_digest: data[268..316].try_into().unwrap(),
123                report_id: data[316..348].try_into().unwrap(),
124                report_id_ma: data[348..380].try_into().unwrap(),
125                reported_tcb: u64::from_le_bytes(data[380..388].try_into().unwrap()),
126                chip_id: data[388..452].try_into().unwrap(),
127                signature: data[672..1184].try_into().unwrap(),
128            })
129        }
130    }
131}

133/// TDX attestation quote parser
134#[cfg(feature = "cpu-tee")]
135pub mod tdx {
136    use super::*;
137
138    #[derive(Debug, Clone, Serialize, Deserialize)]
139    pub struct TdxQuote {
140        pub version: u16,
141        pub attestation_key_type: u16,
142        pub tee_type: u32,
143        pub reserved: [u8; 4],
144        pub vendor_id: [u8; 16],
145        pub user_data: [u8; 20],
146        pub report_data: [u8; 64],
147    }
148
149    impl TdxQuote {
150        pub fn parse(data: &[u8]) -> Result<Self> {
151            if data.len() < 584 {
152                return Err(PqcError::AttestationError("Invalid TDX quote size".into()));
153            }
154
155            Ok(Self {
156                version: u16::from_le_bytes([data[0], data[1]]),
157                attestation_key_type: u16::from_le_bytes([data[2], data[3]]),
158                tee_type: u32::from_le_bytes([data[4], data[5], data[6], data[7]]),
159                reserved: data[8..12].try_into().unwrap(),
160                vendor_id: data[12..28].try_into().unwrap(),
161                user_data: data[28..48].try_into().unwrap(),
162                report_data: data[48..112].try_into().unwrap(),
163            })
164        }
165    }
